Cognitive Dual-Query Strategy with Randomized-Variable-Uncertainty#

  9 import numpy as np
 10 from matplotlib import pyplot as plt, animation
 11 from sklearn.datasets import make_blobs
 12
 13 from skactiveml.utils import MISSING_LABEL
 14 from skactiveml.visualization import (
 15     plot_stream_training_data,
 16     plot_stream_decision_boundary,
 17 )
 18
 19 from skactiveml.classifier import ParzenWindowClassifier
 20 from skactiveml.stream import CognitiveDualQueryStrategyRanVarUn
 21
 22 random_state = np.random.RandomState(0)
 23 init_size = 0
 24 # Build a dataset.
 25 X, y_true = make_blobs(
 26     n_samples=10 + init_size,
 27     n_features=1,
 28     centers=[[0], [-3], [1], [2], [-0.5]],
 29     cluster_std=0.7,
 30     random_state=random_state,
 31 )
 32 y_true = y_true % 2
 33 X_init = X[:init_size]
 34 y_init = y_true[:init_size]
 35 X_stream = X[init_size:]
 36 y_stream = y_true[init_size:]
 37
 38
 39 # Initialise the classifier.
 40 clf = ParzenWindowClassifier(classes=[0, 1], random_state=random_state)
 41 # Initialise the query strategy.
 42 qs = CognitiveDualQueryStrategyRanVarUn(budget=0.2)
 43 plot_step = 5
 44
 45
 46 X_train = []
 47 X_train.extend(X_init)
 48 y_train = []
 49 y_train.extend(y_init)
 50 classes = np.unique(y_true)
 51
 52
 53 # Preparation for plotting.
 54 fig, ax = plt.subplots()
 55
 56 feature_bound = [[0, len(X)], [min(X), max(X)]]
 57 ax.set_xlim(0, len(X))
 58 ax.set_ylim(bottom=min(X), top=max(X))
 59 artists = []
 60
 61 X_train = []
 62 X_train.extend(X_init)
 63 y_train = []
 64 y_train.extend(y_init)
 65
 66 queried_indices = [True] * len(y_init)
 67
 68 predictions_list = []
 69
 70 for t_x, (x_t, y_t) in enumerate(zip(X_stream, y_stream)):
 71
 72     X_cand = x_t.reshape([1, -1])
 73     y_cand = y_t
 74     clf.fit(X_train, y_train)
 75     # Check whether to sample the instance or not
 76     sampled_indices, utilities = qs.query(
 77         candidates=X_cand, clf=clf, return_utilities=True
 78     )
 79     budget_manager_param_dict = {"utilities": utilities}
 80     # Update the query strategy and budget_manager to calculate the right budget
 81     qs.update(candidates=X_cand, queried_indices=sampled_indices)
 82     # Label the queried instances.
 83     X_train.append(x_t)
 84     if len(sampled_indices):
 85         y_train.append(y_t)
 86     else:
 87         y_train.append(MISSING_LABEL)
 88
 89     queried_indices.append(len(sampled_indices) > 0)
 90
 91     # Plot the labeled data.
 92     if t_x % plot_step == 0:
 93
 94         coll_old = list(ax.collections)
 95         ax, predictions_list = plot_stream_decision_boundary(
 96             ax, t_x, plot_step, clf, X, predictions_list, res=3
 97         )
 98         data_lines = plot_stream_training_data(
 99             ax, X_train, y_train, queried_indices, classes, feature_bound
100         )
101
102         title_string = (
103             f"Decision boundary after {t_x} new instances \n"
104             + f"with utility: {utilities[0]: .4f} "
105             + f"budget is {sum(queried_indices) / (t_x + 1):.4f}"
106         )
107         title = ax.text(
108             x=0.5,
109             y=1.05,
110             s=title_string,
111             size=plt.rcParams["axes.titlesize"],
112             ha="center",
113             transform=ax.transAxes,
114         )
115         coll_new = list(ax.collections)
116         coll_new.append(title)
117
118         artists.append(
119             [x for x in coll_new if (x not in coll_old)] + data_lines
120         )
121
122 ani = animation.ArtistAnimation(fig, artists, interval=500, blit=True)
