Dual strategy for Active Learning#

 6 import numpy as np
 7 from matplotlib import pyplot as plt, animation
 8 from sklearn.datasets import make_blobs
 9
10 from skactiveml.utils import MISSING_LABEL, is_labeled, simple_batch
11 from skactiveml.visualization import plot_decision_boundary, \
12     plot_contour_for_samples
13
14 from sklearn.linear_model import LogisticRegression
15 from sklearn.mixture import GaussianMixture
16 from skactiveml.classifier import SklearnClassifier
17 from skactiveml.pool import UncertaintySampling
18
19 random_state = np.random.RandomState(0)
20
21 # Build a dataset.
22 X, y_true = make_blobs(n_samples=10, n_features=2,
23                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
24                        cluster_std=.7, random_state=random_state)
25 y_true = y_true % 2
26 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
27
28 # Initialise the classifier.
29 clf = SklearnClassifier(LogisticRegression(), classes=np.unique(y_true))
30 # Initialise the query strategy.
31 qs = UncertaintySampling(method='least_confident', random_state=random_state)
32 gmm = GaussianMixture(init_params='kmeans', n_components=5)
33 gmm.fit(X)
34 density = np.exp(gmm.score_samples(X))
35 delta = 0.1
36 u_max = -np.inf
37 switching_point = False
38
39 # Preparation for plotting.
40 fig, ax = plt.subplots()
41 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
42 artists = []
43
44 # The active learning cycle:
45 n_cycles = 2
46 for c in range(n_cycles):
47     # Fit the classifier.
48     clf.fit(X, y)
49
50     # Get labeled instances.
51     X_labeled = X[is_labeled(y)]
52
53     # Query the next instance/s.
54     if not switching_point:
55         # DWUS
56         query_idx, utils = qs.query(X=X, y=y, clf=clf, utility_weight=density, return_utilities=True)
57         utilities = utils[0]
58         switching_point = utilities[query_idx[0]] - u_max < delta
59         u_max = utilities[query_idx[0]]
60         strategy = "DWUS"
61     else:
62         # DWUS + US
63         utils_US = qs.query(X=X, y=y, clf=clf, return_utilities=True)[1][0]
64         err = np.nanmean(utils_US)
65         utilities = (1-err)*utils_US + err*density
66         query_idx = simple_batch(utilities, random_state)
67         strategy = "DWUS + US"
68
69     # Plot the labeled data.
70     coll_old = list(ax.collections)
71     title = ax.text(
72         0.5, 1.05, f"Decision boundary after acquring {c} labels with {strategy}",
73         size=plt.rcParams["axes.titlesize"], ha="center",
74         transform=ax.transAxes
75     )
76     ax = plot_contour_for_samples(X, utilities, feature_bound=feature_bound,
77                                   res=31, ax=ax, replace_nan=None)
78     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
79                zorder=2)
80     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
81                marker=".", s=300)
82     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
83
84     coll_new = list(ax.collections)
85     coll_new.append(title)
86     artists.append([x for x in coll_new if (x not in coll_old)])
87
88     # Label the queried instances.
89     y[query_idx] = y_true[query_idx]
90
91 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)

References:

The implementation of this strategy is based on Donmez et al.1 and Nguyen and Smeulders2.

1

Pinar Donmez, Jaime G Carbonell, and Paul N Bennett. Dual strategy active learning. In European Conference on Machine Learning, 116–127. Springer, 2007.

2

Hieu T Nguyen and Arnold Smeulders. Active learning using pre-clustering. In International Conference on Machine Learning, 79. 2004.
