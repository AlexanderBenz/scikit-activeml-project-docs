Value of Information on Unlabeled Samples#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from skactiveml.classifier import ParzenWindowClassifier
17 from skactiveml.pool import ValueOfInformationEER
18
19 random_state = np.random.RandomState(0)
20
21 # Build a dataset.
22 X, y_true = make_blobs(n_samples=10, n_features=2,
23                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
24                        cluster_std=.7, random_state=random_state)
25 y_true = y_true % 2
26 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
27
28 # Initialise the classifier.
29 clf = ParzenWindowClassifier(classes=[0, 1], random_state=random_state)
30 # Initialise the query strategy.
31 qs = ValueOfInformationEER(consider_unlabeled=True,
32                       consider_labeled=False,
33                       candidate_to_labeled=True,
34                       subtract_current=True,
35                       normalize=True)
36
37
38 # Preparation for plotting.
39 fig, ax = plt.subplots()
40 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
41 artists = []
42
43 # The active learning cycle:
44 n_cycles = 2
45 for c in range(n_cycles):
46     # Fit the classifier.
47     clf.fit(X, y)
48
49     # Get labeled instances.
50     X_labeled = X[labeled_indices(y)]
51
52     # Query the next instance/s.
53     query_idx = qs.query(X=X, y=y, clf=clf, ignore_partial_fit=True)
54
55     # Plot the labeled data.
56     coll_old = list(ax.collections)
57     title = ax.text(
58         0.5, 1.05, f"Decision boundary after acquring {c} labels",
59         size=plt.rcParams["axes.titlesize"], ha="center",
60         transform=ax.transAxes
61     )
62     ax = plot_utilities(qs, X=X, y=y, clf=clf, ignore_partial_fit=True,
63                         candidates=None, res=3,
64                         feature_bound=feature_bound, ax=ax)
65     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
66                zorder=2)
67     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
68                marker=".", s=300)
69     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
70
71     coll_new = list(ax.collections)
72     coll_new.append(title)
73     artists.append([x for x in coll_new if (x not in coll_old)])
74
75     # Label the queried instances.
76     y[query_idx] = y_true[query_idx]
77
78 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
