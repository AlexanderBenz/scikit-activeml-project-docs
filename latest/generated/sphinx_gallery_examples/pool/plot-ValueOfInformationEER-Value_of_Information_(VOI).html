Value of Information#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from skactiveml.classifier import ParzenWindowClassifier
17 from skactiveml.pool import ValueOfInformationEER
18
19 random_state = np.random.RandomState(0)
20
21 # Build a dataset.
22 X, y_true = make_blobs(n_samples=10, n_features=2,
23                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
24                        cluster_std=.7, random_state=random_state)
25 y_true = y_true % 2
26 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
27
28 # Initialise the classifier.
29 clf = ParzenWindowClassifier(classes=[0, 1], random_state=random_state)
30 # Initialise the query strategy.
31 qs = ValueOfInformationEER(consider_unlabeled=True,
32                       consider_labeled=True,
33                       candidate_to_labeled=True,
34                       subtract_current=True)
35
36
37 # Preparation for plotting.
38 fig, ax = plt.subplots()
39 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
40 artists = []
41
42 # The active learning cycle:
43 n_cycles = 2
44 for c in range(n_cycles):
45     # Fit the classifier.
46     clf.fit(X, y)
47
48     # Get labeled instances.
49     X_labeled = X[labeled_indices(y)]
50
51     # Query the next instance/s.
52     query_idx = qs.query(X=X, y=y, clf=clf, ignore_partial_fit=True)
53
54     # Plot the labeled data.
55     coll_old = list(ax.collections)
56     title = ax.text(
57         0.5, 1.05, f"Decision boundary after acquring {c} labels",
58         size=plt.rcParams["axes.titlesize"], ha="center",
59         transform=ax.transAxes
60     )
61     ax = plot_utilities(qs, X=X, y=y, clf=clf, ignore_partial_fit=True,
62                         candidates=None, res=3,
63                         feature_bound=feature_bound, ax=ax)
64     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
65                zorder=2)
66     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
67                marker=".", s=300)
68     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
69
70     coll_new = list(ax.collections)
71     coll_new.append(title)
72     artists.append([x for x in coll_new if (x not in coll_old)])
73
74     # Label the queried instances.
75     y[query_idx] = y_true[query_idx]
76
77 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)

References:

The implementation of this strategy is based on Kapoor et al.1.

Ashish Kapoor, Eric Horvitz, and Sumit Basu. Selective supervision: guiding supervised learning with decision-theoretic active learning. In IJCAI, volume 7, 877–882. 2007.
