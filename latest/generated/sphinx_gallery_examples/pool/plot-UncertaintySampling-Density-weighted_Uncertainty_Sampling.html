Density-weighted Uncertainty Sampling#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from sklearn.linear_model import LogisticRegression
17 from sklearn.mixture import GaussianMixture
18 from skactiveml.classifier import SklearnClassifier
19 from skactiveml.pool import UncertaintySampling
20
21 random_state = np.random.RandomState(0)
22
23 # Build a dataset.
24 X, y_true = make_blobs(n_samples=10, n_features=2,
25                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
26                        cluster_std=.7, random_state=random_state)
27 y_true = y_true % 2
28 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
29
30 # Initialise the classifier.
31 clf = SklearnClassifier(LogisticRegression(), classes=np.unique(y_true))
32 # Initialise the query strategy.
33 qs = UncertaintySampling(method='least_confident', random_state=random_state)
34 gmm = GaussianMixture(init_params='kmeans', n_components=5)
35 gmm.fit(X)
36 density = np.exp(gmm.score_samples(X))
37
38 # Preparation for plotting.
39 fig, ax = plt.subplots()
40 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
41 artists = []
42
43 # The active learning cycle:
44 n_cycles = 2
45 for c in range(n_cycles):
46     # Fit the classifier.
47     clf.fit(X, y)
48
49     # Get labeled instances.
50     X_labeled = X[labeled_indices(y)]
51
52     # Query the next instance/s.
53     query_idx = qs.query(X=X, y=y, clf=clf, utility_weight=density)
54
55     # Plot the labeled data.
56     coll_old = list(ax.collections)
57     title = ax.text(
58         0.5, 1.05, f"Decision boundary after acquring {c} labels",
59         size=plt.rcParams["axes.titlesize"], ha="center",
60         transform=ax.transAxes
61     )
62     ax = plot_utilities(qs, X=X, y=y, clf=clf, utility_weight=density,
63                         candidates=unlabeled_indices(y), replace_nan=None, res=3,
64                         feature_bound=feature_bound, ax=ax)
65     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
66                zorder=2)
67     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
68                marker=".", s=300)
69     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
70
71     coll_new = list(ax.collections)
72     coll_new.append(title)
73     artists.append([x for x in coll_new if (x not in coll_old)])
74
75     # Label the queried instances.
76     y[query_idx] = y_true[query_idx]
77
78 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)

References:

The implementation of this strategy is based on Donmez et al.1 and Nguyen and Smeulders2.

1

Pinar Donmez, JaimeG Carbonell, and PaulN Bennett. Dual strategy active learning. In European Conference on Machine Learning, 116127. Springer, 2007.

2

HieuT Nguyen and Arnold Smeulders. Active learning using pre-clustering. In International Conference on Machine Learning, 79. 2004.
