Discriminative Active Learning#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from skactiveml.classifier import ParzenWindowClassifier
17 from skactiveml.pool import DiscriminativeAL
18 from sklearn import clone
19
20 random_state = np.random.RandomState(0)
21
22 # Build a dataset.
23 X, y_true = make_blobs(n_samples=10, n_features=2,
24                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
25                        cluster_std=.7, random_state=random_state)
26 y_true = y_true % 2
27 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
28
29 # Initialise the classifier.
30 clf = ParzenWindowClassifier(classes=[0, 1], random_state=random_state)
31 # Initialise the query strategy.
32 qs = DiscriminativeAL()
33 discriminator=clone(clf)
34
35 # Preparation for plotting.
36 fig, ax = plt.subplots()
37 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
38 artists = []
39
40 # The active learning cycle:
41 n_cycles = 2
42 for c in range(n_cycles):
43     # Fit the classifier.
44     clf.fit(X, y)
45
46     # Get labeled instances.
47     X_labeled = X[labeled_indices(y)]
48
49     # Query the next instance/s.
50     query_idx = qs.query(X=X, y=y, discriminator=discriminator)
51
52     # Plot the labeled data.
53     coll_old = list(ax.collections)
54     title = ax.text(
55         0.5, 1.05, f"Decision boundary after acquring {c} labels",
56         size=plt.rcParams["axes.titlesize"], ha="center",
57         transform=ax.transAxes
58     )
59     ax = plot_utilities(qs, X=X, y=y, discriminator=discriminator,
60                         candidates=None, res=3,
61                         feature_bound=feature_bound, ax=ax)
62     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
63                zorder=2)
64     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
65                marker=".", s=300)
66     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
67
68     coll_new = list(ax.collections)
69     coll_new.append(title)
70     artists.append([x for x in coll_new if (x not in coll_old)])
71
72     # Label the queried instances.
73     y[query_idx] = y_true[query_idx]
74
75 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
