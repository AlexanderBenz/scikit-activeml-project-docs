Regression based Kullback Leibler Divergence Maximization#

  9 import numpy as np
 10 from matplotlib import pyplot as plt, animation
 11 from scipy.stats import uniform
 12
 13 from skactiveml.utils import MISSING_LABEL, labeled_indices, is_labeled
 14
 15 from skactiveml.regressor import NICKernelRegressor
 16 from skactiveml.pool import KLDivergenceMaximization
 17
 18 random_state = np.random.RandomState(0)
 19
 20
 21 def true_function(X_):
 22     return (X_**3 + 2 * X_**2 + X_ - 1).flatten()
 23
 24 n_samples = 10
 25 X = np.concatenate(
 26         [uniform.rvs(0, 1.5, 9 * n_samples // 10, random_state=random_state),
 27          uniform.rvs(1.5, 0.5, n_samples // 10, random_state=random_state)]
 28     ).reshape(-1, 1)
 29
 30 noise = np.vectorize(lambda x : random_state.rand() * 1.5 if x < 1
 31                                 else random_state.rand() * 0.5)
 32
 33 # Build a dataset.
 34 y_true = true_function(X) + noise(X).flatten()
 35 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
 36 X_test = np.linspace(0, 2, num=3).reshape(-1, 1)
 37
 38 # Initialise the classifier.
 39 reg = NICKernelRegressor(random_state=random_state, metric_dict={'gamma': 15.0})
 40 # Initialise the query strategy.
 41 qs = KLDivergenceMaximization(
 42     random_state=random_state,
 43     integration_dict_target_val={
 44         "method": "gauss_hermite",
 45         "n_integration_samples": 5,
 46     },
 47 )
 48
 49
 50 # Preparation for plotting.
 51 fig, (ax_1, ax_2) = plt.subplots(2, 1, sharex=True)
 52 artists = []
 53
 54 # The active learning cycle:
 55 n_cycles = 2
 56 for c in range(n_cycles):
 57     # Fit the classifier.
 58     reg.fit(X, y)
 59
 60     # Get labeled instances.
 61     X_labeled = X[labeled_indices(y)]
 62
 63     # Query the next instance/s.
 64     query_idx = qs.query(X=X, y=y, reg=reg)
 65
 66     # Plot the labeled data.
 67     coll_old = list(ax_1.collections) + list(ax_2.collections)
 68     title = ax_1.text(
 69         0.5,
 70         1.05,
 71         f"Prediction after acquring {c} labels",
 72         size=plt.rcParams["axes.titlesize"],
 73         ha="center",
 74         transform=ax_1.transAxes,
 75     )
 76
 77     _, utilities_test = qs.query(
 78         X=X, y=y, reg=reg, candidates=X_test, return_utilities=True
 79     )
 80
 81     utilities_test = (utilities_test - utilities_test.min()).flatten()
 82     if np.any(utilities_test != utilities_test[0]):
 83         utilities_test /= utilities_test.max()
 84
 85     is_lbld = is_labeled(y)
 86
 87     (utility_line,) = ax_2.plot(X_test, utilities_test, c="green")
 88     utility_fill = plt.fill_between(
 89         X_test.flatten(), utilities_test, color="green", alpha=0.3
 90     )
 91
 92     ax_1.scatter(X[~is_lbld], y_true[~is_lbld], c="lightblue")
 93     ax_1.scatter(X[is_lbld], y[is_lbld], c="orange")
 94
 95     y_pred = reg.predict(X_test)
 96     (prediction_line,) = ax_1.plot(X_test, y_pred, c="black")
 97
 98     coll_new = list(ax_1.collections) + list(ax_2.collections)
 99     coll_new.append(title)
100     artists.append(
101         [x for x in coll_new if (x not in coll_old)]
102         + [utility_line, utility_fill, prediction_line]
103     )
104
105     # Label the queried instances.
106     y[query_idx] = y_true[query_idx]
107
108 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
