Monte-Carlo EER with Misclassification-Loss#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from skactiveml.classifier import ParzenWindowClassifier
17 from skactiveml.pool import MonteCarloEER
18
19 random_state = np.random.RandomState(0)
20
21 # Build a dataset.
22 X, y_true = make_blobs(n_samples=10, n_features=2,
23                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
24                        cluster_std=.7, random_state=random_state)
25 y_true = y_true % 2
26 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
27
28 # Initialise the classifier.
29 clf = ParzenWindowClassifier(classes=[0, 1], random_state=random_state)
30 # Initialise the query strategy.
31 qs = MonteCarloEER(method='misclassification_loss')
32
33
34 # Preparation for plotting.
35 fig, ax = plt.subplots()
36 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
37 artists = []
38
39 # The active learning cycle:
40 n_cycles = 2
41 for c in range(n_cycles):
42     # Fit the classifier.
43     clf.fit(X, y)
44
45     # Get labeled instances.
46     X_labeled = X[labeled_indices(y)]
47
48     # Query the next instance/s.
49     query_idx = qs.query(X=X, y=y, clf=clf, ignore_partial_fit=True)
50
51     # Plot the labeled data.
52     coll_old = list(ax.collections)
53     title = ax.text(
54         0.5, 1.05, f"Decision boundary after acquring {c} labels",
55         size=plt.rcParams["axes.titlesize"], ha="center",
56         transform=ax.transAxes
57     )
58     ax = plot_utilities(qs, X=X, y=y, clf=clf, ignore_partial_fit=True,
59                         candidates=None, res=3,
60                         feature_bound=feature_bound, ax=ax)
61     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
62                zorder=2)
63     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
64                marker=".", s=300)
65     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
66
67     coll_new = list(ax.collections)
68     coll_new.append(title)
69     artists.append([x for x in coll_new if (x not in coll_old)])
70
71     # Label the queried instances.
72     y[query_idx] = y_true[query_idx]
73
74 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)

References:

The implementation of this strategy is based on Roy and McCallum1.

1

Nicholas Roy and Andrew McCallum. Toward optimal active learning through monte carlo estimation of error reduction. In Proceedings of the International Conference on Machine Learning, 441–448. 2001.
