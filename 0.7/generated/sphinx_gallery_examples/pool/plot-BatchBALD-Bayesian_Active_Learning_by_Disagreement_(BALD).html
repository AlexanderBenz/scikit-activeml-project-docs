Bayesian Active Learning by Disagreement#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary
15
16 from sklearn.gaussian_process import GaussianProcessClassifier
17 from sklearn.ensemble import BaggingClassifier
18 from skactiveml.classifier import SklearnClassifier
19 from skactiveml.pool import BatchBALD
20
21 random_state = np.random.RandomState(0)
22
23 # Build a dataset.
24 X, y_true = make_blobs(n_samples=10, n_features=2,
25                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
26                        cluster_std=.7, random_state=random_state)
27 y_true = y_true % 2
28 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
29
30 # Initialise the classifier.
31 clf = SklearnClassifier(BaggingClassifier(
32     SklearnClassifier(GaussianProcessClassifier(), random_state=random_state),
33     random_state=random_state),
34     classes=np.unique(y_true),
35     random_state=random_state
36 )
37 # Initialise the query strategy.
38 qs = BatchBALD(random_state=random_state)
39
40
41 # Preparation for plotting.
42 fig, ax = plt.subplots()
43 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
44 artists = []
45
46 # The active learning cycle:
47 n_cycles = 2
48 for c in range(n_cycles):
49     # Fit the classifier.
50     clf.fit(X, y)
51
52     # Get labeled instances.
53     X_labeled = X[labeled_indices(y)]
54
55     # Query the next instance/s.
56     query_idx = qs.query(X=X, y=y, ensemble=clf)
57
58     # Plot the labeled data.
59     coll_old = list(ax.collections)
60     title = ax.text(
61         0.5, 1.05, f"Decision boundary after acquring {c} labels",
62         size=plt.rcParams["axes.titlesize"], ha="center",
63         transform=ax.transAxes
64     )
65     ax = plot_utilities(qs, X=X, y=y, ensemble=clf,
66                         replace_nan=None, candidates=X[unlabeled_indices(y)], res=3,
67                         feature_bound=feature_bound, ax=ax)
68     ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
69                zorder=2)
70     ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
71                marker=".", s=300)
72     ax = plot_decision_boundary(clf, feature_bound, ax=ax)
73
74     coll_new = list(ax.collections)
75     coll_new.append(title)
76     artists.append([x for x in coll_new if (x not in coll_old)])
77
78     # Label the queried instances.
79     y[query_idx] = y_true[query_idx]
80
81 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
