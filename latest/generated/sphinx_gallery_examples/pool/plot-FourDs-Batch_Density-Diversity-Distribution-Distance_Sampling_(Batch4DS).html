Batch Density-Diversity-Distribution-Distance Sampling#

 9 import numpy as np
10 from matplotlib import pyplot as plt, animation
11 from sklearn.datasets import make_blobs
12
13 from skactiveml.utils import MISSING_LABEL, labeled_indices, unlabeled_indices
14 from skactiveml.visualization import plot_utilities, plot_decision_boundary, \
15     plot_contour_for_samples
16
17 from skactiveml.classifier import MixtureModelClassifier
18 from skactiveml.pool import FourDs
19 from sklearn.mixture import GaussianMixture
20
21 random_state = np.random.RandomState(0)
22
23 # Build a dataset.
24 X, y_true = make_blobs(n_samples=10, n_features=2,
25                        centers=[[0, 1], [-3, .5], [-1, -1], [2, 1], [1, -.5]],
26                        cluster_std=.7, random_state=random_state)
27 y_true = y_true % 2
28 y = np.full(shape=y_true.shape, fill_value=MISSING_LABEL)
29
30 # Initialise the classifier.
31 clf = MixtureModelClassifier(classes=[0, 1], mixture_model=GaussianMixture(n_components=5))
32 # Initialise the query strategy.
33 qs = FourDs()
34
35
36 # Preparation for plotting.
37 fig, axs = plt.subplots(2, 2, constrained_layout=True)
38 feature_bound = [[min(X[:, 0]), min(X[:, 1])], [max(X[:, 0]), max(X[:, 1])]]
39 artists = [[] for j in range(2)]
40
41 # The active learning cycle:
42 n_cycles = 2
43 for c in range(n_cycles):
44     # Fit the classifier.
45     clf.fit(X, y)
46
47     # Get labeled instances.
48     X_labeled = X[labeled_indices(y)]
49
50     # Query the next instance/s.
51     query_idx, utilities = qs.query(X=X, y=y, clf=clf, batch_size=4, return_utilities=True)
52
53     # Plot the labeled data.
54     for i, ax in enumerate(axs.flatten()):
55         coll_old = list(ax.collections)
56         plot_contour_for_samples(X, utilities[i], res=3,
57                                  feature_bound=feature_bound, replace_nan=None, ax=ax)
58         ax.scatter(X[:, 0], X[:, 1], c=y_true, cmap="coolwarm", marker=".",
59                    zorder=2)
60         ax.scatter(X_labeled[:, 0], X_labeled[:, 1], c="grey", alpha=.8,
61                    marker=".", s=300)
62         ax = plot_decision_boundary(clf, feature_bound, ax=ax)
63         ax.set_title(f"Batch {c+1}, Utilities[{i}]")
64
65         for x in ax.collections:
66             if x not in coll_old:
67                 artists[c].append(x)
68
69     # Label the queried instances.
70     y[query_idx] = y_true[query_idx]
71
72 ani = animation.ArtistAnimation(fig, artists, interval=1000, blit=True)
